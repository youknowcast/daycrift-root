export const BuildYoutubeLink = ({ url, start }) => {
  const yturl = `${url}?start=${start}`;
  return <iframe width="800" height="400" src={yturl} frameborder="0" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>
}

<!-- 本ページ専用の h4 タグを作成する -->
export const CustomH4 = ({ idx, time }) => {
  const address = `skill_${parseInt(idx) + 1}`;
  const showableTime = `${Math.floor(time/60)}:${time%60}`
  return <> <h4> {skills[idx]} <small> ※ {showableTime} から </small> </h4><a id={address}></a></>
}
